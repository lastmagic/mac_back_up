        for(int i=0; i<5; i++){
            if(check[i]==true&&check[i+5]==true&&check[i+10]==true&&check[i+15]==true&&check[i+20]==true) {//技肺
                return true;
            }
        }

           int i=4;
            if(check[i]==true&&check[i+4]==true&&check[i+8]==true&&check[i+12]==true&&check[i+16]==true){//剧狼规氢 措阿急
                return true;
            }

        for(i=0; i<1;i++){
            if(check[i]==true&&check[i+6]==true&&check[i+12]==true&&check[i+18]==true&&check[i+24]==true){//澜狼规氢 措阿急
                return true;
            }
        }
        for(i=0; i<5; i++){

            if(check[5*i]==true&&check[5*i+1]==true&&check[5*i+2]==true&&check[5*i+3]==true&&check[5*i+4]==true){//啊肺
                return true;
            }
        }
        return false;
    }
